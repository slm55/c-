1-Бөлім: Математикалық Жүрек

Бұл функция экранның төменгі бөлігінде орналасатын жүректің нүктелерін есептейді:

void setupHeart(sf::ConvexShape& heart) {
    heart.setPointCount(40);
    for (int i = 0; i < 40; ++i) {
        float angle = (2.f * 3.14159f * i) / 40.f;
        float x = 16.f * std::pow(std::sin(angle), 3);
        float y = -(13.f * std::cos(angle) - 5.f * std::cos(2 * angle) - 2.f * std::cos(3 * angle) - std::cos(4 * angle));
        heart.setPoint(i, {x * 8.f, y * 8.f});
    }
    heart.setFillColor(sf::Color(255, 30, 30));
    heart.setPosition({400.f, 480.f});
}

2-Бөлім: Негізгі Бағдарлама (main)

Мұнда терезе құру, қаріпті жүктеу және анимация циклі орналасқан:

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "For Mom");
    window.setFramerateLimit(60);
    sf::Clock clock;

    sf::Font font;
    if (!font.openFromFile("/System/Library/Fonts/Palatino.ttc")) return -1;

    sf::Text title(font, "8 NAURYZ QUTTY BOLSYN!", 45);
    title.setFillColor(sf::Color::Magenta);
    
    // Орталықтандыру логикасын осында жазыңыз...

    sf::ConvexShape heart;
    setupHeart(heart);

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
        }

        float time = clock.getElapsedTime().asSeconds();
        
        // Жүректің соғу анимациясы
        float pulse = 1.0f + std::sin(time * 3.f) * 0.15f;
        heart.setScale({pulse, pulse});

        window.clear(sf::Color(40, 5, 30));
        window.draw(title);
        window.draw(heart);
        window.display();
    }
    return 0;
}
