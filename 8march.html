#include <SFML/Graphics.hpp> #include <cmath> #include <vector> const sf::Vector2f SCREEN_SIZE = {800.f, 600.f}; const sf::Vector2f CENTER = {400.f, 280.f}; void centerText(sf::Text& text, float y) { auto bounds = text.getGlobalBounds(); text.setOrigin({bounds.size.x / 2.f, 0.f}); text.setPosition({SCREEN_SIZE.x / 2.f, y}); } void setupHeart(sf::ConvexShape& heart) { heart.setPointCount(40); for (int i = 0; i < 40; ++i) { float angle = (2.f * 3.14159f * i) / 40.f; float x = 16.f * std::pow(std::sin(angle), 3); float y = -(13.f * std::cos(angle) - 5.f * std::cos(2 * angle) - 2.f * std::cos(3 * angle) - std::cos(4 * angle)); heart.setPoint(i, {x * 8.f, y * 8.f}); } heart.setFillColor(sf::Color(255, 30, 30)); heart.setOutlineThickness(2.f); heart.setOutlineColor(sf::Color::White); heart.setPosition({SCREEN_SIZE.x / 2.f, 480.f}); } int main() { sf::RenderWindow window(sf::VideoMode({(unsigned int)SCREEN_SIZE.x, (unsigned int)SCREEN_SIZE.y}), "Happy 8 March"); window.setFramerateLimit(60); sf::Clock clock; sf::Font font; if (!font.openFromFile("/System/Library/Fonts/Palatino.ttc")) { if (!font.openFromFile("/System/Library/Fonts/Supplemental/Arial.ttf")) return -1; } sf::Text title(font, "8 NAURYZ QUTTY BOLSYN!", 45); title.setFillColor(sf::Color(255, 182, 193)); title.setStyle(sf::Text::Bold); centerText(title, 40.f); sf::Text subTitle(font, "Ayauly ana, merekenizben!", 28); subTitle.setFillColor(sf::Color::White); centerText(subTitle, 100.f); sf::ConvexShape heart; setupHeart(heart); std::vector<sf::CircleShape> petals; for (int i = 0; i < 8; ++i) { sf::CircleShape petal(45.f); petal.setFillColor(sf::Color(255, 105, 180, 180)); petal.setOrigin({45.f, 90.f}); petal.setPosition(CENTER); petals.push_back(petal); } sf::CircleShape flowerCenter(35.f); flowerCenter.setFillColor(sf::Color(255, 215, 0)); flowerCenter.setOrigin({35.f, 35.f}); flowerCenter.setPosition(CENTER); while (window.isOpen()) { while (const auto event = window.pollEvent()) { if (event->is<sf::Event::Closed>()) window.close(); } float time = clock.getElapsedTime().asSeconds(); for (size_t i = 0; i < petals.size(); ++i) { petals[i].setRotation(sf::degrees(i * 45.f + time * 40.f)); } float pulse = 1.0f + std::sin(time * 3.f) * 0.15f; heart.setScale({pulse, pulse}); window.clear(sf::Color(40, 5, 30)); window.draw(title); window.draw(subTitle); window.draw(heart); for (const auto& p : petals) window.draw(p); window.draw(flowerCenter); window.display(); } return 0; }
