Нұсқаулық: Кодты оқып, IDE-ге теріңіз. 
#include <SFML/Graphics.hpp>
#include <vector>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "SFML Mini Paint");
    window.setFramerateLimit(120); // Сурет салу бірқалыпты болуы үшін FPS-ті көбейтеміз

    // Барлық салынған "нүктелерді" сақтайтын жиым
    std::vector<sf::CircleShape> brushStrokes;
    
    // Қылқаламның баптаулары
    float brushSize = 5.f;
    sf::Color brushColor = sf::Color::Cyan;

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
            
            // "C" пернесін басқанда экранды тазалау
            if (const auto* keyEvent = event->getIf<sf::Event::KeyPressed>()) {
                if (keyEvent->code == sf::Keyboard::Key::C) {
                    brushStrokes.clear();
                }
            }
        }

        // Нақты уақытта тышқанның сол жақ батырмасы басылып тұрғанын тексеру
        if (sf::Mouse::isButtonPressed(sf::Mouse::Button::Left)) {
            sf::Vector2i mousePos = sf::Mouse::getPosition(window);
            
            // Жаңа нүкте жасау
            sf::CircleShape dot(brushSize);
            dot.setFillColor(brushColor);
            // Нүктенің ортасын тышқанның ұшына дәл келтіру
            dot.setOrigin({brushSize, brushSize});
            dot.setPosition({static_cast<float>(mousePos.x), static_cast<float>(mousePos.y)});
            
            brushStrokes.push_back(dot);
        }

        // --- СЫЗУ ---
        window.clear(sf::Color::Black);
        
        for (const auto& dot : brushStrokes) {
            window.draw(dot);
        }
        
        window.display();
    }
    return 0;
}
