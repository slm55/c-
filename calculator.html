Нұсқаулық: Тінтуірмен басқа терезеге ауысуға болады. Бірақ iMac-те ⌘ + Shift басылса, код жасырылады. 
#include <SFML/Graphics.hpp>
#include <string>
#include <vector>
#include <iostream>

// Түйме құрылымы: SFML 3.0-де sf::Text үшін шрифт алдын ала керек
struct Button {
    sf::RectangleShape shape;
    sf::Text text;
    std::string value;

    Button(const sf::Font& font, const std::string& label, sf::Vector2f size, sf::Vector2f pos)
        : text(font, label) 
    {
        shape.setSize(size);
        shape.setPosition(pos);
        shape.setFillColor(sf::Color(60, 60, 70));

        text.setCharacterSize(30);
        text.setFillColor(sf::Color::White);

        // Мәтінді түйменің ортасына дәлдеу
        sf::FloatRect textBounds = text.getLocalBounds();
        text.setOrigin({textBounds.size.x / 2.f, textBounds.size.y / 2.f});
        text.setPosition(pos + sf::Vector2f(size.x / 2.f, size.y / 2.f));

        value = label;
    }
};

int main() {
    // 1. Терезе құру
    sf::RenderWindow window(sf::VideoMode({400, 550}), "SFML 3.0 Calculator");
    window.setFramerateLimit(60);

    // 2. Шрифт жүктеу
    sf::Font font;
    if (!font.openFromFile("/System/Library/Fonts/Supplemental/Arial.ttf")) {
        std::cerr << "Шрифт табылмады!" << std::endl;
        return -1;
    }

    // 3. Калькулятор айнымалылары
    std::string currentStr = "";
    double firstNum = 0;
    std::string operation = "";

    // 4. Дисплей мәтіні
    sf::Text display(font, "0", 50);
    display.setPosition({20.f, 20.f});
    display.setFillColor(sf::Color::White);

    // 5. Түймелерді жасау
    std::vector<Button> buttons;
    std::vector<std::string> labels = {
        "7", "8", "9", "/",
        "4", "5", "6", "*",
        "1", "2", "3", "-",
        "C", "0", "=", "+"
    };

    for (int i = 0; i < 16; ++i) {
        sf::Vector2f size = {80.f, 80.f};
        sf::Vector2f pos = {20.f + (i % 4) * 90.f, 120.f + (i / 4) * 90.f};
        buttons.emplace_back(font, labels[i], size, pos);
    }

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();

            if (const auto* mouseEvent = event->getIf<sf::Event::MouseButtonPressed>()) {
                sf::Vector2f mPos = { (float)mouseEvent->position.x, (float)mouseEvent->position.y };

                for (auto& btn : buttons) {
                    if (btn.shape.getGlobalBounds().contains(mPos)) {
                        std::string val = btn.value;

                        if (val == "C") {
                            currentStr = "";
                            operation = "";
                            firstNum = 0;
                        }
                        else if (val == "+" || val == "-" || val == "*" || val == "/") {
                            if (!currentStr.empty()) {
                                firstNum = std::stod(currentStr);
                                operation = val;
                                currentStr = "";
                            }
                        }
                        else if (val == "=") {
                            if (!currentStr.empty() && !operation.empty()) {
                                double secondNum = std::stod(currentStr);
                                double result = 0;

                                if (operation == "+") result = firstNum + secondNum;
                                else if (operation == "-") result = firstNum - secondNum;
                                else if (operation == "*") result = firstNum * secondNum;
                                else if (operation == "/" && secondNum != 0) result = firstNum / secondNum;

                                currentStr = std::to_string(result);
                                currentStr.erase(currentStr.find_last_not_of('0') + 1, std::string::npos);
                                if (currentStr.back() == '.') currentStr.pop_back();

                                operation = "";
                            }
                        }
                        else {
                            if (currentStr.length() < 12) currentStr += val;
                        }
                        display.setString(currentStr.empty() ? "0" : currentStr);
                    }
                }
            }
        }

        window.clear(sf::Color(30, 30, 40));
        window.draw(display);
        for (const auto& btn : buttons) {
            window.draw(btn.shape);
            window.draw(btn.text);
        }
        window.display();
    }
    return 0;
}
