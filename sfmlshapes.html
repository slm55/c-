SFML 3.0: Графика және Қозғалыс

Заманауи C++ стиліндегі графика, фигуралар және анимация негіздері

1. Координаталар жүйесі (Coordinates)

SFML-де экранның жоғарғы сол жақ нүктесі (0, 0) болып табылады. Математикадан айырмашылығы — Y осі төмен қарай өседі.

  • (0, 0): Экранның сол жақ жоғарғы бұрышы.
  • X осі: Оңға қарай жылжыған сайын өседі.
  • Y осі: Төмен қарай жылжыған сайын өседі.
Егер сіз объектіні төмен түсіргіңіз келсе, Y мәнін арттыруыңыз керек.

3. Төртбұрыш (RectangleShape)

Бұл мысалда біз экранның ортасында орналасқан, жиектері бар көк тіктөртбұрыш жасаймыз.

#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "Rectangle 3.0 Example");

    // Төртбұрыш құру ({ені, биіктігі})
    sf::RectangleShape rect({200.f, 100.f});
    
    // Appearance (Сырт келбеті)
    rect.setFillColor(sf::Color::Blue);
    rect.setOutlineThickness(10.f);
    rect.setOutlineColor(sf::Color::White);
    rect.setPosition({300.f, 250.f});

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
        }

        window.clear();
        window.draw(rect);
        window.display();
    }
    return 0;
}

4. Шеңберлер мен Көпбұрыштар (CircleShape)

CircleShape арқылы нүктелер санын өзгерту арқылы үшбұрыш немесе алтыбұрыш жасауға болады.

#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "Shapes 3.0 Test");

    // 1. Нағыз шеңбер (Радиусы 50)
    sf::CircleShape circle(50.f);
    circle.setFillColor(sf::Color::Red);
    circle.setPosition({100, 100});

    // 2. Үшбұрыш (3 нүктелі шеңбер)
    sf::CircleShape triangle(60.f, 3);
    triangle.setFillColor(sf::Color::Green);
    triangle.setPosition({300, 100});

    // 3. Алтыбұрыш (6 нүктелі шеңбер)
    sf::CircleShape hex(60.f, 6);
    hex.setFillColor(sf::Color::Yellow);
    hex.setPosition({500, 100});

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
        }

        window.clear();
        window.draw(circle);
        window.draw(triangle);
        window.draw(hex);
        window.display();
    }
    return 0;
}

5. Ерікті фигуралар (ConvexShape)

Өз қалауыңызша нүктелер қою арқылы кез келген пішінді салу:

#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "Custom Convex Shape");

    sf::ConvexShape star;
    star.setPointCount(5); 
    star.setPoint(0, {50, 0});
    star.setPoint(1, {100, 40});
    star.setPoint(2, {80, 100});
    star.setPoint(3, {20, 100});
    star.setPoint(4, {0, 40});
    
    star.setFillColor(sf::Color::Magenta);
    star.setPosition({350, 250});

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
        }

        window.clear();
        window.draw(star);
        window.display();
    }
    return 0;
}

6. Мәтін және Жүйелік Қаріптер (Text)

Мәтін шығару үшін Apple немесе Windows жүйесіндегі қаріпті пайдаланудың толық жолы:

#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "System Font Text");

    sf::Font font;
    if (!font.openFromFile("/System/Library/Fonts/Supplemental/Arial.ttf")) { return -1; }

    sf::Text text(font, "Salem SFML 3.0!", 50);
    text.setFillColor(sf::Color::Cyan);
    text.setStyle(sf::Text::Bold);
    text.setPosition({200, 250});

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
        }

        window.clear();
        window.draw(text);
        window.display();
    }
    return 0;
}

7. Қозғалыс (Movement) және Анимация

Қозғалыс — әр кадр сайын позицияны өзгерту. SFML 3.0-де бұрыштарды sf::degrees() арқылы береміз.

#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "Movement & Animation");
    window.setFramerateLimit(60);

    sf::CircleShape player(30.f);
    player.setFillColor(sf::Color::Green);
    player.setPosition({400, 300});

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
        }

        // Қозғалыс логикасы
        float speed = 5.f;
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::W)) player.move({0, -speed});
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::S)) player.move({0, speed});
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::A)) player.move({-speed, 0});
        if (sf::Keyboard::isKeyPressed(sf::Keyboard::Key::D)) player.move({speed, 0});

        // Анимация: Аздап айналдыру
        player.rotate(sf::degrees(1.0f));

        window.clear();
        window.draw(player);
        window.display();
    }
    return 0;
}

8. Қорытынды МИНИ-ЖОБА

Барлық білімді біріктіретін толық бағдарлама:

#include <SFML/Graphics.hpp>

int main() {
    sf::RenderWindow window(sf::VideoMode({800, 600}), "Final Project 3.0");
    window.setFramerateLimit(60);

    // Объект құру
    sf::CircleShape star(50.f, 6);
    star.setFillColor(sf::Color::Yellow);
    star.setOutlineThickness(5.f);
    star.setOutlineColor(sf::Color::Red);
    star.setOrigin({50.f, 50.f});
    star.setPosition({400.f, 300.f});

    // Мәтін
    sf::Font font;
    font.loadFromFile("arial.ttf");
    sf::Text msg(font, "Arrows to move, Space to scale", 20);
    msg.setPosition({10, 10});

    while (window.isOpen()) {
        while (const auto event = window.pollEvent()) {
            if (event->is<sf::Event::Closed>()) window.close();
            
            // Event арқылы пернені тексеру
            if (const auto* key = event->getIf<sf::Event::KeyPressed>()) {
                if (key->code == sf::Keyboard::Key::Space) {
                    star.setScale({1.5f, 1.5f});
                }
            }
        }

        // Тұрақты айналу анимациясы
        star.rotate(sf::degrees(2.0f));

        window.clear(sf::Color(20, 20, 20));
        window.draw(star);
        window.draw(msg);
        window.display();
    }
    return 0;
}
